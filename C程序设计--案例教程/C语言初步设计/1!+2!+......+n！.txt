int fun(int n)
{
	int i=1,j=1;
	while(i<=n)
	{
		j*=i;
		i++;
	}
//	printf("the result is %d\n",j);
	
}
main()
{
	int sum=0;
	int cnt=0;
	int n;
	printf("Plaese input n:");
	scanf("%d",&n);
	for(cnt=0;cnt<=n;cnt++)
	{
		sum+=fun(cnt);
	}
	printf("the result is %d\n",sum);
	
	return 0;
}
