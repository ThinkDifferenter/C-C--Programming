#include<stdio.h>
#include<stdlib.h>
#include<time.h>
#define N 10

void charu (int *a,int n);
main()
{
	int arr[N],i;
	
	srand(time(NULL));
	
	for(i=0;i<N;i++)
	{
		arr[i]=rand()/1000+100;
	}	
	printf("排序之前:\n");
	for(i=0;i<N;i++)
	{
		printf("%d\t",arr[i]);
	}
	printf("\n");
	printf("排序之后:\n");
	
	charu(arr,N);
	
	for(i=0;i<N;i++)
	{
		printf("%d\t",arr[i]);
	}
	printf("\n");
	return 0;
}
void charu (int *a,int n)
{
	int i,j,t;
	
	for(i=1;i<n;i++)
	{
		t=a[i];
		j=i-1;
		
		while(j>=0&&t<a[j])
		{
			a[j+1]=a[j];
			j--;
		}
		a[j+1]=t;
	}
}
