#include<stdio.h>
#include<stdlib.h>
#include<time.h>
#define N 10

void maopao (int *a,int n);
main()
{
	int arr[N],i;
	
	srand(time(NULL));
	
	for(i=0;i<N;i++)
	{
		arr[i]=rand()/1000+100;
	}	
	printf("排序之前:\n");
	for(i=0;i<N;i++)
	{
		printf("%d\t",arr[i]);
	}
	printf("\n");
	printf("排序之后:\n");
	maopao(arr,N);
	
	for(i=0;i<N;i++)
	{
		printf("%d\t",arr[i]);
	}
	printf("\n");
	return 0;
}
void maopao(int *a,int n)
{
	int i,j,t;
	
	for(i=0;i<n;i++)
	{
		for(j=n-1;j>=i;j--)
		{
			if(a[j-1]>a[j])
			{
				t=a[j-1];
				a[j-1]=a[j];
				a[j]=t;
			}
		}
	}
}
